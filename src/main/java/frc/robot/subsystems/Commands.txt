package frc.robot.subsystems;
import java.util.Map;

import edu.wpi.first.wpilibj.shuffleboard.BuiltInLayouts;
import edu.wpi.first.wpilibj.shuffleboard.BuiltInWidgets;
import edu.wpi.first.wpilibj.shuffleboard.Shuffleboard;
import edu.wpi.first.wpilibj.shuffleboard.ShuffleboardLayout;
import edu.wpi.first.wpilibj.shuffleboard.ShuffleboardTab;
import edu.wpi.first.wpilibj2.command.SubsystemBase;
import frc.robot.commands.auto.Pick;
import frc.robot.commands.auto.Place;
public class Commands extends SubsystemBase  {
    private final ShuffleboardLayout elevatorCommands = Shuffleboard.getTab("Commands")
    .getLayout("ArmCommands", BuiltInLayouts.kList)
    .withSize(2, 2)
    .withProperties(Map.of("Label position", "HIDDEN")); // hide labels for commands
    
    public Commands(){

    }

    @Override
    public void periodic(){
        elevatorCommands.add(new Pick());
        elevatorCommands.add(new Place());
    }
}
